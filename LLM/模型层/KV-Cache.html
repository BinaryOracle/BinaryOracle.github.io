def generate_text(model, tokenizer, prompt, max_new_tokens=50, eos_token_id=198):
    model.eval()
    input_ids = tokenizer.encode(prompt, return_tensors="pt")
    past_key_values = None
    output_ids = input_ids.clone()

    with torch.no_grad():
        for step in range(max_new_tokens):
            outputs = model(
                input_ids=input_ids,
                past_key_values=past_key_values,
                use_cache=True

class GPT2Block(GradientCheckpointingLayer):

    def forward(
        self,
        hidden_states: Optional[tuple[torch.FloatTensor]],
        past_key_value: Optional[Cache] = None,
        cache_position: Optional[torch.LongTensor] = None,
        attention_mask: Optional[torch.FloatTensor] = None,
        use_cache: Optional[bool] = False,
    ) -> Union[tuple[torch.Tensor], Optional[tuple[torch.Tensor, tuple[torch.FloatTensor, ...]]]]:
        
        # 1. å½’ä¸€åŒ–
        residual = hidden_states
        hidden_states = self.ln_1(hidden_states)
        
        # 2. è‡ªæ³¨æ„åŠ›è¿ç®—
        attn_output, self_attn_weights = self.attn(
            hidden_states,
            past_key_value=past_key_value,
            cache_position=cache_position,
            attention_mask=attention_mask,
            use_cache=use_cache,
        )
        
        # 3. residual connection
        hidden_states = attn_output + residual

        # 4. å½’ä¸€åŒ– + MLP +  residual connection
        residual = hidden_states
        hidden_states = self.ln_2(hidden_states)
        feed_forward_hidden_states = self.mlp(hidden_states)
        hidden_states = residual + feed_forward_hidden_states

        return hidden_states

class GPT2Attention(nn.Module):

    def __init__(self, config, is_cross_attention=False, layer_idx=None):
        self.c_attn = Conv1D(3 * self.embed_dim, self.embed_dim) # è¾“å…¥ç»´åº¦: (batch,seq_len,embed_dim) , å˜æ¢åçš„è¾“å‡ºç»´åº¦: (batch,seq_len,3*embed_dim)
        self.c_proj = Conv1D(self.embed_dim, self.embed_dim)

    def forward(
        self,
        hidden_states: Optional[tuple[torch.FloatTensor]],
        past_key_value: Optional[Cache] = None,
        cache_position: Optional[torch.LongTensor] = None,
        attention_mask: Optional[torch.FloatTensor] = None,
    ) -> tuple[Union[torch.Tensor, tuple[torch.Tensor]], ...]:
        # 1. ä¸€ç»´å·ç§¯è¿›è¡Œçº¿æ€§å˜æ¢å’Œå‡ç»´ï¼Œç„¶ååˆ‡åˆ†æˆqueryï¼Œkeyï¼Œvalue
        query_states, key_states, value_states = self.c_attn(hidden_states).split(self.split_size, dim=2)

        # 2. (batch,seq_len,-1,head_dim) , head_dim æ˜¯å¤šå¤´è‡ªæ³¨æ„åŠ›ä¸­æ¯ä¸ªå¤´åˆ‡åˆ†åˆ°çš„ç»´åº¦ 
        shape_q = (*query_states.shape[:-1], -1, self.head_dim)
        shape_kv = (*key_states.shape[:-1], -1, self.head_dim)
        
        # 3. ç»´åº¦ç»Ÿä¸€: (batch,heads,seq_len,head_dim)
        query_states = query_states.view(shape_q).transpose(1, 2)
        key_states = key_states.view(shape_kv).transpose(1, 2)
        value_states = value_states.view(shape_kv).transpose(1, 2)
         
        # 4. KV Cache ä¸ä¸ºç©º 
        if past_key_value is not None:
            # 4.1 cache_position è®°å½•å½“å‰è¯å¯¹åº”è¾“å…¥è¯åºåˆ—ä¸­çš„ç´¢å¼•
            cache_kwargs = {"cache_position": cache_position}
            # 4.2 å°†å½“å‰è¯çš„keyå’Œvalè¿›è¡Œç¼“å­˜ï¼Œæ ¹æ®æ‰€åœ¨GPTBlockå±‚çº§(layer_idxè¯´æ˜)ï¼Œå’Œä½äºè¯åºåˆ—çš„ç´¢å¼•(cache_kwargsè¯´æ˜),æ’å…¥å¯¹åº”å±‚çš„listç¼“å­˜ä¸­å»ï¼ŒåŒæ—¶è¿”å›å¯¹åº”çš„keyå’Œval list
            key_states, value_states = past_key_value.update(
                key_states, value_states, self.layer_idx, cache_kwargs=cache_kwargs
            )

        # 5. è¿›è¡Œç»å…¸çš„å¤šå¤´è‡ªæ³¨æ„åŠ›è¿ç®—(ä¸å±•å¼€ç»†èŠ) 
        attn_output, attn_weights = attention_interface(
            self,
            query_states, # å½“å‰è¾“å…¥è¯çš„query
            key_states,   # cache key list + è¾“å…¥è¯çš„key
            value_states,  # cache val list + è¾“å…¥è¯çš„val
            attention_mask, # padding mask
            dropout=self.attn_dropout.p if self.training else 0.0,
        )
          
        attn_output = attn_output.reshape(*attn_output.shape[:-2], -1).contiguous()
        attn_output = self.c_proj(attn_output)
        attn_output = self.resid_dropout(attn_output)

        return attn_output, attn_weights
