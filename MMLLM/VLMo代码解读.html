    å°†è¾“å…¥çš„å›¾åƒåˆ‡åˆ†æˆå° patchï¼Œå¹¶é€šè¿‡å·ç§¯æ˜ å°„åˆ°æŒ‡å®šçš„ embedding ç»´åº¦ã€‚
    è¿™æ˜¯ Vision Transformer (ViT) ä¸­å¸¸ç”¨çš„å›¾åƒåµŒå…¥æ–¹æ³•ã€‚

    def __init__(

        patch_size=16,         # æ¯ä¸ªpatchçš„é«˜åº¦å’Œå®½åº¦ï¼ˆé»˜è®¤16x16ï¼‰
        in_chans=3,            # è¾“å…¥é€šé“æ•°ï¼Œå½©è‰²å›¾åƒé€šå¸¸ä¸º3

        patch_size = to_2tuple(patch_size)

        # ä¿å­˜ patch è¡Œåˆ—æ•°ï¼Œç”¨äºä½ç½®ç¼–ç æˆ–å…¶ä»–å¤„ç†

        # å®šä¹‰å·ç§¯å±‚ï¼Œå°†å›¾åƒåˆ‡åˆ†æˆ patch å¹¶æ˜ å°„åˆ° embedding ç»´åº¦

            in_chans,
            embed_dim,
            kernel_size=patch_size,
            stride=patch_size,
            bias=False if no_patch_embed_bias else True,
        )

    def forward(self, x):
        B, C, H, W = x.shape
        # æ£€æŸ¥è¾“å…¥å›¾åƒå°ºå¯¸æ˜¯å¦ä¸åˆå§‹åŒ–å°ºå¯¸åŒ¹é…

            f"Input image size ({H}*{W}) doesn't match model ({self.img_size[0]}*{self.img_size[1]})."
        # å·ç§¯æ˜ å°„ï¼Œå°†å›¾åƒåˆ‡åˆ†æˆ patch å¹¶ç”Ÿæˆ embedding

        """
        å°†è¾“å…¥çš„å›¾åƒå¼ é‡ _x è½¬æ¢ä¸ºè§†è§‰ token embeddingã€‚
        æ­¥éª¤åŒ…æ‹¬ï¼š
        1. patch embedding
        2. å±•å¹³å¹¶è°ƒæ•´ç»´åº¦

class Block(nn.Module):
    def __init__(
        self,
        dim,                # è¾“å…¥ç‰¹å¾ç»´åº¦
        num_heads,          # å¤šå¤´æ³¨æ„åŠ›çš„å¤´æ•°
        mlp_ratio=4.0,      # MLP éšè—å±‚ç»´åº¦ä¸è¾“å…¥ç»´åº¦çš„æ¯”ä¾‹
        qkv_bias=False,     # QKV æ˜¯å¦ä½¿ç”¨åç½®
        qk_scale=None,      # QK ç¼©æ”¾å› å­ï¼ˆè¦†ç›–é»˜è®¤ head_dim ** -0.5ï¼‰
        drop=0.0,           # Dropout æ¦‚ç‡
        attn_drop=0.0,      # æ³¨æ„åŠ›æƒé‡çš„ Dropout æ¦‚ç‡
        drop_path=0.0,      # Stochastic Depth æ¦‚ç‡
        act_layer=nn.GELU,  # æ¿€æ´»å‡½æ•°ç±»å‹
        norm_layer=nn.LayerNorm, # å½’ä¸€åŒ–å±‚ç±»å‹
        with_vlffn=False,   # æ˜¯å¦ä½¿ç”¨è·¨æ¨¡æ€ MLPï¼ˆVision-Language Feed-Forward Networkï¼‰
        layer_scale_init_values=0.1, # LayerScale åˆå§‹åŒ–å€¼
        max_text_len=40,    # æœ€å¤§æ–‡æœ¬åºåˆ—é•¿åº¦
    ):
        super().__init__()

        # ç¬¬ä¸€ä¸ª LayerNormï¼ˆä½œç”¨äºæ³¨æ„åŠ›ä¹‹å‰ï¼‰
        self.norm1 = norm_layer(dim)

        # å¤šå¤´æ³¨æ„åŠ›æœºåˆ¶
        self.attn = Attention(
            dim,
            num_heads=num_heads,
            qkv_bias=qkv_bias,
            qk_scale=qk_scale,
            attn_drop=attn_drop,
            proj_drop=drop,
        )

        # DropPathï¼ˆéšæœºä¸¢å¼ƒæ•´ä¸ªæ®‹å·®åˆ†æ”¯ï¼‰æˆ–æ’ç­‰æ˜ å°„
        self.drop_path = DropPath(drop_path) if drop_path > 0.0 else nn.Identity()

        # ç¬¬äºŒé˜¶æ®µçš„ LayerNormï¼ˆé’ˆå¯¹æ–‡æœ¬å’Œå›¾åƒåˆ†åˆ«æœ‰ç‹¬ç«‹çš„å½’ä¸€åŒ–å±‚ï¼‰
        self.norm2_text = norm_layer(dim)
        self.norm2_imag = norm_layer(dim)

        # MLP éšè—å±‚ç»´åº¦
        mlp_hidden_dim = int(dim * mlp_ratio)

        # æ–‡æœ¬æ¨¡æ€çš„ MLP
        self.mlp_text = Mlp(
            in_features=dim,
            hidden_features=mlp_hidden_dim,
            act_layer=act_layer,
            drop=drop,
        )

        # å›¾åƒæ¨¡æ€çš„ MLP
        self.mlp_imag = Mlp(
            in_features=dim,
            hidden_features=mlp_hidden_dim,
            act_layer=act_layer,
            drop=drop,
        )

        # è·¨æ¨¡æ€ MLPï¼ˆä»…åœ¨ with_vlffn=True æ—¶ä½¿ç”¨ï¼‰
        self.mlp_vl = None
        if with_vlffn:
            self.mlp_vl = Mlp(
                in_features=dim,
                hidden_features=mlp_hidden_dim,
                act_layer=act_layer,
                drop=drop,
            )
            self.norm2_vl = norm_layer(dim)

        # LayerScale å‚æ•°ï¼ˆgamma_1 ä½œç”¨äºæ³¨æ„åŠ›åˆ†æ”¯ï¼Œgamma_2 ä½œç”¨äº MLP åˆ†æ”¯ï¼‰
        self.gamma_1 = (
            nn.Parameter(layer_scale_init_values * torch.ones((dim)), requires_grad=True)
            if layer_scale_init_values is not None else 1.0
        )
        self.gamma_2 = (
            nn.Parameter(layer_scale_init_values * torch.ones((dim)), requires_grad=True)
            if layer_scale_init_values is not None else 1.0
        )

        # æœ€å¤§æ–‡æœ¬é•¿åº¦ï¼ˆåœ¨æ‹†åˆ†å¤šæ¨¡æ€è¾“å…¥æ—¶ä½¿ç”¨ï¼‰
        self.max_text_len = max_text_len

    def forward(self, x, mask=None, modality_type=None, relative_position_bias=None):
        """
        Args:
            x: è¾“å…¥ç‰¹å¾ [B, L, C]
            mask: æ³¨æ„åŠ›æ©ç ï¼ˆå¯é€‰ï¼‰
            modality_type: è¾“å…¥æ¨¡æ€ç±»å‹ ("image", "text", æˆ– None è¡¨ç¤ºå¤šæ¨¡æ€ï¼‰
            relative_position_bias: ç›¸å¯¹ä½ç½®ç¼–ç åç½®
        """

        # ====== æ³¨æ„åŠ›å­å±‚ï¼ˆå¸¦æ®‹å·®è¿æ¥ + LayerScale + DropPathï¼‰======
        x = x + self.drop_path(
            self.gamma_1 * self.attn(
                self.norm1(x),
                mask=mask,
                relative_position_bias=relative_position_bias
            )
        )

        # ====== å‰é¦ˆç½‘ç»œå­å±‚ï¼ˆæ ¹æ®æ¨¡æ€ç±»å‹é€‰æ‹©ä¸åŒ MLPï¼‰======
        if modality_type == "image":
            # ä»…å›¾åƒæ¨¡æ€
            x = x + self.drop_path(self.gamma_2 * self.mlp_imag(self.norm2_imag(x)))
        elif modality_type == "text":
            # ä»…æ–‡æœ¬æ¨¡æ€
            x = x + self.drop_path(self.gamma_2 * self.mlp_text(self.norm2_text(x)))
        else:
            # å¤šæ¨¡æ€æƒ…å†µ
            if self.mlp_vl is None:
                # åˆ†å¼€å¤„ç†æ–‡æœ¬å’Œå›¾åƒåºåˆ—
                x_text = x[:, : self.max_text_len]   # å‰ max_text_len ä¸ºæ–‡æœ¬
                x_imag = x[:, self.max_text_len :]   # å‰©ä½™éƒ¨åˆ†ä¸ºå›¾åƒ
                x_text = x_text + self.drop_path(self.gamma_2 * self.mlp_text(self.norm2_text(x_text)))
                x_imag = x_imag + self.drop_path(self.gamma_2 * self.mlp_imag(self.norm2_imag(x_imag)))
                # åˆå¹¶å›ä¸€ä¸ªåºåˆ—
                x = torch.cat([x_text, x_imag], dim=1)
            else:
                # è·¨æ¨¡æ€ MLP

        # QK ç¼©æ”¾å› å­ï¼Œé»˜è®¤ä¸º 1/sqrt(head_dim)ï¼Œé˜²æ­¢ç‚¹ç§¯ç»“æœè¿‡å¤§

        # çº¿æ€§æ˜ å°„å¾—åˆ° Qã€Kã€Vï¼ˆåœ¨è¿™é‡Œä¸€æ¬¡æ€§è®¡ç®—ï¼‰
