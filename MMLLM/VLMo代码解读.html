        attn_drop=0.0,      # æ³¨æ„åŠ›æƒé‡çš„ Dropout æ¦‚ç‡

        # QK ç¼©æ”¾å› å­ï¼Œé»˜è®¤ä¸º 1/sqrt(head_dim)ï¼Œé˜²æ­¢ç‚¹ç§¯ç»“æœè¿‡å¤§

            self.v_bias = None

    def forward(self, x, mask=None, relative_position_bias=None):
        """
        Args:
            x: è¾“å…¥å¼ é‡ (B, N, C)ï¼Œ
               B=æ‰¹å¤§å°ï¼ŒN=åºåˆ—é•¿åº¦ï¼ŒC=é€šé“æ•°(embedding dim)
            mask: æ³¨æ„åŠ›æ©ç  (B, N)ï¼Œç”¨äºå±è”½æ— æ•ˆä½ç½®

        """
        B, N, C = x.shape  # å–å‡ºæ‰¹å¤§å°ã€åºåˆ—é•¿åº¦ã€é€šé“æ•°

        # å¤„ç† Qã€Kã€V åç½®

            qkv_bias = torch.cat((
                self.q_bias,
                torch.zeros_like(self.v_bias, requires_grad=False),
                self.v_bias

        # çº¿æ€§æ˜ å°„å¾—åˆ° Qã€Kã€Vï¼ˆåœ¨è¿™é‡Œä¸€æ¬¡æ€§è®¡ç®—ï¼‰

        # æ‹†åˆ† Qã€Kã€V

        # ç¼©æ”¾ Q

        # æ³¨æ„åŠ›åŠ æƒ Vï¼Œç„¶åè¿˜åŸç»´åº¦ä¸º (B, N, C)

        # æŠ•å½±ç»“æœ Dropout

        return x
