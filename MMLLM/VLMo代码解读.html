        # QK ç¼©æ”¾å› å­ï¼Œé»˜è®¤ä¸º 1/sqrt(head_dim)ï¼Œé˜²æ­¢ç‚¹ç§¯ç»“æœè¿‡å¤§

        """
        Args:

            mask: æ³¨æ„åŠ›æ©ç  (B, N)ï¼Œç”¨äºå±è”½æ— æ•ˆä½ç½®

        B, N, C = x.shape  # å–å‡ºæ‰¹å¤§å°ã€åºåˆ—é•¿åº¦ã€é€šé“æ•°

        # å¤„ç† Qã€Kã€V åç½®

            qkv_bias = torch.cat((
                self.q_bias,
                torch.zeros_like(self.v_bias, requires_grad=False),
                self.v_bias

        # çº¿æ€§æ˜ å°„å¾—åˆ° Qã€Kã€Vï¼ˆåœ¨è¿™é‡Œä¸€æ¬¡æ€§è®¡ç®—ï¼‰

        # æ‹†åˆ† Qã€Kã€V

        # ç¼©æ”¾ Q

        # æŠ•å½±ç»“æœ Dropout

        return x

def compute_itc(pl_module, batch, aggregate=True):
    # 1. åˆ†åˆ«å¯¹å›¾åƒå’Œæ–‡æœ¬åšå‰å‘æ¨ç†ï¼Œæå– ITC æ‰€éœ€ç‰¹å¾

    logit_vl_scale = pl_module.logit_vl_scale.exp().mean()

    # ========== å¤„ç†åˆ†å¸ƒå¼è®­ç»ƒä¸‹çš„è´Ÿæ ·æœ¬æ‰©å±• ==========
    if aggregate:
        ...   
    else:

        logits_per_image = logit_scale * image_features @ text_features.t()
        logits_per_text = logit_scale * text_features @ image_features.t()

    itc_loss = (
        F.cross_entropy(logits_per_image.float(), ground_truth)
        + F.cross_entropy(logits_per_text.float(), ground_truth)
    ) / 2

    # 10. è®¡ç®—è·¨æ¨¡æ€ VL-FFN ITC loss (ä»…åœ¨åˆ†å¸ƒå¼ç¯å¢ƒä¸‹å¯ç”¨)

    ) / 2

    ret = {

    with torch.no_grad():
