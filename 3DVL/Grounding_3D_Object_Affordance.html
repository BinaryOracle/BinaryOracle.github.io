class Affordance_Decoder(nn.Module):
    def __init__(self, emb_dim, proj_dim):
        super().__init__()
        self.emb_dim = emb_dim
        self.proj_dim = proj_dim
        self.cross_atten = Cross_Attention(emb_dim = self.emb_dim, proj_dim = self.proj_dim)
        self.fusion = nn.Sequential(
            nn.Conv1d(2*self.emb_dim, self.emb_dim, 1, 1),
            nn.BatchNorm1d(self.emb_dim),
            nn.ReLU()
        )

    def forward(self, query, key, value):
        '''
        query: [B, N_p + N_i, C]   -> spatial_feature (query)
        key:   [B, N_l, C]         -> instructional_feature (key)
        value: [B, N_l, C]         -> semantic_feature (value)
        '''
        B, _, C = query.size()

        # è°ƒæ•´ key å’Œ value çš„å½¢çŠ¶ä¸º [B, C, N_l]
        key = key.view(B, C, -1)  # [B, C, N_l]
        value = value.view(B, C, -1)  # [B, C, N_l]

        # ä½¿ç”¨ cross attention è·å–ä¸¤ä¸ªæ³¨æ„åŠ›åŠ æƒç»“æœ

        # å°†ä¸¤ä¸ªæ³¨æ„åŠ›è¾“å‡ºæ‹¼æ¥åœ¨ä¸€èµ·

        # ä½¿ç”¨ Conv1D èåˆé€šé“ä¿¡æ¯

        # è°ƒæ•´è¾“å‡ºæ ¼å¼ä¸º [B, N_p + N_i, C]

class Cross_Attention(nn.Module):
    def __init__(self, emb_dim, proj_dim):
        """
        å¤šæ¨¡æ€äº¤å‰æ³¨æ„åŠ›æ¨¡å—ï¼ˆCross-Attention Moduleï¼‰ï¼Œ
        ç”¨äºèåˆæ¥è‡ªè¯­è¨€æ¨¡å‹çš„ä¸åŒè¯­ä¹‰ä¿¡æ¯ï¼Œå¢å¼ºç©ºé—´ç‰¹å¾è¡¨è¾¾ã€‚

        Args:
            emb_dim: è¾“å…¥ç‰¹å¾ç»´åº¦ï¼ˆembedding dimensionï¼‰ï¼Œä¾‹å¦‚ LLM çš„ hidden sizeï¼ˆå¦‚ 4096ï¼‰
            proj_dim: æŠ•å½±ç»´åº¦ï¼Œç”¨äºé™ä½è®¡ç®—å¤æ‚åº¦ï¼Œåœ¨ attention ä¸­ä½¿ç”¨
        """
        super().__init__()
        self.emb_dim = emb_dim
        self.proj_dim = proj_dim

        # å®šä¹‰æŠ•å½±å±‚ï¼Œå°†è¾“å…¥æ˜ å°„åˆ°ä½ç»´ç©ºé—´ä»¥è¿›è¡Œ attention è®¡ç®—
        self.proj_q = nn.Linear(self.emb_dim, proj_dim)   # query æŠ•å½±
        self.proj_sk = nn.Linear(self.emb_dim, proj_dim)  # sub key æŠ•å½±
        self.proj_sv = nn.Linear(self.emb_dim, proj_dim)  # sub value æŠ•å½±
        self.proj_ek = nn.Linear(self.emb_dim, proj_dim)  # scene key æŠ•å½±
        self.proj_ev = nn.Linear(self.emb_dim, proj_dim)  # scene value æŠ•å½±

        # ç¼©æ”¾å› å­ï¼Œç”¨äº attention åˆ†æ•°å½’ä¸€åŒ–
        self.scale = self.proj_dim ** (-0.5)

        # å±‚å½’ä¸€åŒ–ï¼ˆLayerNormï¼‰ï¼Œç”¨äºç¨³å®šè®­ç»ƒè¿‡ç¨‹
        self.layernorm = nn.LayerNorm(self.emb_dim)

    def forward(self, obj, sub, scene):
        """
        æ‰§è¡Œäº¤å‰æ³¨æ„åŠ›æœºåˆ¶ï¼Œèåˆä¸åŒæ¥æºçš„ä¿¡æ¯ï¼š
        - obj: ç©ºé—´ç‰¹å¾ï¼ˆspatial featureï¼‰ï¼Œä½œä¸º queryï¼›
        - sub: æŒ‡ä»¤ç†è§£ç‰¹å¾ï¼ˆinstructional featureï¼‰ï¼Œä½œä¸ºç¬¬ä¸€ä¸ª attention çš„ key å’Œ valueï¼›
        - scene: è§†è§‰è¯­ä¹‰ç‰¹å¾ï¼ˆsemantic featureï¼‰ï¼Œä½œä¸ºç¬¬äºŒä¸ª attention çš„ key å’Œ valueï¼›

        Args:
            obj: [B, N_p + HW, C] â†’ spatial_featureï¼ˆquery æ¥æºï¼‰
            sub: [B, HW, C] â†’ instructional_featureï¼ˆkey/value æ¥æºä¹‹ä¸€ï¼‰
            scene: [B, HW, C] â†’ semantic_featureï¼ˆkey/value æ¥æºä¹‹äºŒï¼‰

        Returns:
            I_1: ç»è¿‡ attention åŠ æƒåçš„è¾“å‡ºï¼ˆç¬¬ä¸€åˆ†æ”¯ï¼‰
            I_2: ç»è¿‡ attention åŠ æƒåçš„è¾“å‡ºï¼ˆç¬¬äºŒåˆ†æ”¯ï¼‰
        """

        B, seq_length, C = obj.size()  # è·å– batch size å’Œé€šé“ç»´åº¦

        # å°†è¾“å…¥åˆ†åˆ«æŠ•å½±åˆ°ä½ç»´ç©ºé—´ï¼Œä¾¿äºåç»­ attention è®¡ç®—
        query = self.proj_q(obj)                   # [B, N_q, proj_dim]
        s_key = self.proj_sk(sub)                  # [B, N_i, proj_dim]
        s_value = self.proj_sv(sub)                # [B, N_i, proj_dim]

        e_key = self.proj_ek(scene)                # [B, N_e, proj_dim]
        e_value = self.proj_ev(scene)              # [B, N_e, proj_dim]

        # ç¬¬ä¸€ä¸ª cross attentionï¼šä½¿ç”¨ sub çš„ key å’Œ value å¢å¼º query
        atten_I1 = torch.bmm(query, s_key.mT) * self.scale  # [B, N_q, N_i]
        atten_I1 = atten_I1.softmax(dim=-1)                 # softmax å½’ä¸€åŒ–
        I_1 = torch.bmm(atten_I1, s_value)                  # [B, N_q, proj_dim]

        # ç¬¬äºŒä¸ª cross attentionï¼šä½¿ç”¨ scene çš„ key å’Œ value å¢å¼º query
        atten_I2 = torch.bmm(query, e_key.mT) * self.scale  # [B, N_q, N_e]
        atten_I2 = atten_I2.softmax(dim=-1)
        I_2 = torch.bmm(atten_I2, e_value)                 # [B, N_q, proj_dim]

        # ä½¿ç”¨æ®‹å·®è¿æ¥ + LayerNorm å¢å¼ºç¨³å®šæ€§
        I_1 = self.layernorm(obj + I_1)  # [B, N_q, emb_dim]
        I_2 = self.layernorm(obj + I_2)  # [B, N_q, emb_dim]

        return I_1, I_2

class Head(nn.Module):
    def __init__(self, additional_channel, emb_dim, N_p, N_raw):
        """
        Head æ¨¡å—ç”¨äºæœ€ç»ˆçš„ 3D å¯æ“ä½œæ€§ï¼ˆaffordanceï¼‰é¢„æµ‹ã€‚
        å®ƒæ¥æ”¶æ¥è‡ªç¼–ç å™¨å’Œè§£ç å™¨çš„ç‰¹å¾ï¼Œå¹¶é€šè¿‡å¤šå°ºåº¦ä¸Šé‡‡æ ·ä¸èåˆï¼Œ
        è¾“å‡ºæ¯ä¸ªç‚¹äº‘ç‚¹çš„ affordance çƒ­å›¾ï¼ˆheatmapï¼‰ï¼Œè¡¨ç¤ºè¯¥ç‚¹æ˜¯å¦å…·æœ‰å¯æ“ä½œæ€§ã€‚

        Args:
            additional_channel: é¢å¤–é€šé“æ•°ï¼Œä¾‹å¦‚æ³•å‘é‡ã€é¢œè‰²ç­‰ä¿¡æ¯

            - æœ€ç»ˆä½¿ç”¨ MLP + Sigmoid è¾“å‡ºæ¯ä¸ªç‚¹çš„ affordance scoreï¼›
        """
        super().__init__()
        
        self.emb_dim = emb_dim
        self.N_p = N_p         # point cloud token æ•°é‡
        self.N_raw = N_raw     # åŸå§‹ç‚¹äº‘æ•°é‡ï¼ˆå¦‚ 2048ï¼‰

        # å¤šå°ºåº¦ä¸Šé‡‡æ ·æ¨¡å—ï¼šPointNetFeaturePropagation

        self.fp3 = PointNetFeaturePropagation(in_channel=512+self.emb_dim, mlp=[768, 512])  
        self.fp2 = PointNetFeaturePropagation(in_channel=832, mlp=[768, 512]) 
        self.fp1 = PointNetFeaturePropagation(in_channel=518+additional_channel, mlp=[512, 512]) 

        # å…¨å±€å¹³å‡æ± åŒ–å±‚ï¼Œå‹ç¼©æ—¶é—´/ç©ºé—´ç»´åº¦

    def forward(self, multi_feature, affordance_feature, encoder_p):
        """
        æ‰§è¡Œ Head æ¨¡å—çš„å‰å‘ä¼ æ’­ï¼Œç”Ÿæˆæœ€ç»ˆçš„ 3D affordance heatmapã€‚

        # è§£åŒ…ç¼–ç å™¨è¾“å‡ºçš„ä¸åŒå±‚çº§ç‰¹å¾
        p_0, p_1, p_2, p_3 = encoder_p

        up_sample = self.fp3(p_2[0], p_3[0], p_2[1], P_align.mT)  # [B, emb_dim, npoint_sa2]

        F_pa_pool = self.pool(F_pa.mT)  # [B, emb_dim, 1]

        affordance = up_sample * F_pa_pool.expand(-1, -1, self.N_raw)  # [B, emb_dim, N_raw]

        out = self.out_head(affordance.mT)  # [B, N_raw, 1]
