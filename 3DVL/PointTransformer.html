        # æ³¨æ„è¿™é‡Œåšäº†â€œé€šé“åˆ†ç»„â€ï¼ˆshare_planesï¼‰ï¼Œå‡å°‘è®¡ç®—é‡

            nn.ReLU(inplace=True),

            nn.BatchNorm1d(mid_planes // share_planes),

    def forward(self, pxo) -> torch.Tensor:

        # p: ç‚¹çš„åæ ‡ (n, 3)
        # x: ç‚¹çš„ç‰¹å¾ (n, c)

        # x_k: (n, nsample, 3+c)ï¼ŒåŒ…å«ç›¸å¯¹åæ ‡å’Œ K ç‰¹å¾
        # x_v: (n, nsample, c)ï¼Œé‚»åŸŸå†…çš„ V ç‰¹å¾

        x_v = pointops.queryandgroup(self.nsample, p, p, x_v, None, o, o, use_xyz=False)

        # åˆ†ç¦»ç›¸å¯¹åæ ‡ p_r å’Œé‚»åŸŸå†…çš„ K ç‰¹å¾
        p_r, x_k = x_k[:, :, 0:3], x_k[:, :, 3:]

        # å°†ç›¸å¯¹åæ ‡ p_r è¾“å…¥ä½ç½®ç¼–ç  MLP Î¸
        # è¿™é‡Œå› ä¸º BatchNorm çš„ç»´åº¦é—®é¢˜ï¼Œéœ€è¦è½¬ç½® (n, nsample, 3) â†” (n, 3, nsample)
        for i, layer in enumerate(self.linear_p):

        # ç»è¿‡ MLP å: (n, nsample, out_planes)

        # æ ¹æ® Eq.(3): w = Î³(Ï†(xi) âˆ’ Ïˆ(xj) + Î´)
        # x_q.unsqueeze(1): (n, 1, c)ï¼Œä¸é‚»åŸŸå¯¹é½
        # p_r reshape åä¸ x_k å¯¹é½åšç›¸åŠ 
        w = x_k - x_q.unsqueeze(1) + p_r.view(
            p_r.shape[0], p_r.shape[1], self.out_planes // self.mid_planes, self.mid_planes
        ).sum(2)  # (n, nsample, c)

        # å°† w è¾“å…¥ Î³ MLP (linear_w)ï¼Œå¾—åˆ°æ³¨æ„åŠ›æƒé‡
        for i, layer in enumerate(self.linear_w):

        w = self.softmax(w)  # (n, nsample, c)

        # æœ€ç»ˆèšåˆ (Eq.(3) ä¸­ Ï(...)*Î±(xj+Î´))

        return x

    @staticmethod
    def forward(ctx, nsample, xyz, new_xyz, offset, new_offset):
        """

        """

            new_xyz = xyz  # å¦‚æœæ²¡æœ‰æŒ‡å®šæŸ¥è¯¢ç‚¹ï¼Œåˆ™å¯¹æ‰€æœ‰ç‚¹è¿›è¡Œè‡ªæŸ¥è¯¢
        assert xyz.is_contiguous() and new_xyz.is_contiguous()
        m = new_xyz.shape[0]  # æŸ¥è¯¢ç‚¹çš„æ•°é‡

        # åˆå§‹åŒ–è¾“å‡ºå¼ é‡ï¼šç´¢å¼•çŸ©é˜µå’Œè·ç¦»çŸ©é˜µ
        idx = torch.zeros((m, nsample), dtype=torch.long)
        dist2 = torch.zeros((m, nsample))

        # æŒ‰batchå¤„ç†æ•°æ®
        start_idx, new_start_idx = 0, 0  # å½“å‰batchçš„èµ·å§‹ç´¢å¼•

            # ç¡®ä¿å½“å‰batchæœ‰æ•°æ®éœ€è¦å¤„ç†

                # æå–å½“å‰batchçš„ç‚¹åæ ‡å’ŒæŸ¥è¯¢ç‚¹åæ ‡

                # ä½¿ç”¨å¹¿æ’­æœºåˆ¶è®¡ç®—åæ ‡å·®: (1,n,3) - (m,1,3) = (m,n,3) - (m,n,3) = (m_batch, n_batch, 3)
                diff = batch_xyz.unsqueeze(0) - batch_new_xyz.unsqueeze(1)  
                # (m_batch, n_batch) - å¹³æ–¹è·ç¦»çŸ©é˜µ
                distances = torch.sum(diff ** 2, dim=-1)

                # è·å–kä¸ªæœ€è¿‘é‚»çš„ç´¢å¼•å’Œè·ç¦»

                # å¦‚æœå®é™…é‚»å±…æ•°é‡å°äºè¦æ±‚çš„nsampleï¼Œè¿›è¡Œå¡«å……

                # å°†å½“å‰batchçš„ç»“æœå­˜å…¥æ€»è¾“å‡ºä¸­ï¼Œæ³¨æ„åŠ ä¸Šå…¨å±€åç§»é‡

            start_idx, new_start_idx = end_idx, new_end_idx
